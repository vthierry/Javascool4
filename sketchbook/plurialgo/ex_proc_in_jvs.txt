void calculer(double prixUnitaire, int quantite) { 
	double remise; 
	double prixTotal; 
	prixTotal = quantite * prixUnitaire;
	if (quantite == 1) {
		remise = 0; 
	}
	else if (quantite <= 3) {
		remise = 0.1; 
	}
	else {
		remise = 0.2; 
	}
	prixTotal = prixTotal - remise * prixTotal; 
	println( "prixTotal : " + prixTotal );  
	println( "remise : " + remise ); 
} 
void main() {
	int quantite; 
	double prixUnitaire; 
	quantite = readInt( "quantite : " ); 
	prixUnitaire = readDouble( "prixUnitaire : " ); 
	calculer(prixUnitaire, quantite); 
}
