
calculer_prixTotal(prixUnitaire, quantite):={
	local remise,prixTotal;
	prixTotal := quantite * prixUnitaire;
	si (quantite == 1) alors
		remise := 0;
	sinon
		si(quantite <= 3) alors
			remise := 0.1;
		sinon
			remise := 0.2;
		fsi;
	fsi;
	prixTotal := prixTotal - remise * prixTotal;
	retourne prixTotal;
};
saisir("quantite : ",quantite); 
saisir("prixUnitaire : ",prixUnitaire); 
prixTotal := calculer_prixTotal(prixUnitaire, quantite);
afficher(prixTotal);
