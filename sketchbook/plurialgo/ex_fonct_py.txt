def calculer_prixTotal(prixUnitaire, quantite) : 
	prixTotal=quantite * prixUnitaire 
	if (quantite == 1) : 
		remise=0 
	elif (quantite <= 3) : 
		remise=0.1 
	else : 
		remise=0.2 
	#end
	prixTotal=prixTotal - remise * prixTotal 
	return prixTotal
#end

prixUnitaire = input("Quel est le prix unitaire en euros ?") 
quantite = input("Quelle est la quantite achetee ?")
prixTotal=calculer_prixTotal(prixUnitaire, quantite) 
print "Le prix total vaut ", prixTotal, " euros"
