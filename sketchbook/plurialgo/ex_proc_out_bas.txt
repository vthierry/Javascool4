Option Explicit

sub calculer(ByVal prixUnitaire as double, ByVal quantite as integer, ByRef remise as double, ByRef prixTotal as double) 
' passage par valeur (ByVal), passage par reference (ByRef)
	prixTotal = quantite * prixUnitaire 
	if (quantite = 1) then 
		remise = 0
	elseif (quantite <= 3) then 
		remise = 0.1 
	else
		remise = 0.2
	end if
	prixTotal = prixTotal - remise * prixTotal 
end function

sub main()
	Dim quantite as integer
	Dim prixUnitaire as double
	Dim prixTotal as double
	Dim remise as double
	quantite = InputBox("quantite : ") 
	prixUnitaire = InputBox("prix unitaire : ") 
	call calculer(prixUnitaire, quantite, remise, prixTotal)	
	MsgBox("prix total : " & prixTotal) 
	MsgBox("remise : " & remise) 
end sub
